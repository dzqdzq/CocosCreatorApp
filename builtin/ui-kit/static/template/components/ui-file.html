VUE JS

<ui-file ref="file" value="{{filepath}}" @confirm="confirm($event.target.value)" ></ui-file> // set value this.$refs.file.value = '{{filepath}}'; // get value const filepath = this.$refs.file.value;

<ui-file id="file" value="{{filepath}}"></ui-file> const element = document.getElementById('#file'); // addEventListener element.addEventListener('confirm', (event)=>{ // get value const filepath = event.target.value; }); // set value element.value = '{{filepath}}'; // get value const filepath = element.value;
